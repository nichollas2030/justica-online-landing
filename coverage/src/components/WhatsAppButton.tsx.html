export const WhatsAppButton = () => {
  // Número do WhatsApp do escritório (substitua pelo número real)
  const whatsappNumber = "5511987654321"; // Exemplo: +55 11 98765-4321
  const message = "Olá! Gostaria de agendar uma consulta jurídica gratuita. Vim através do site.";
  
  const handleWhatsAppClick = () => {
    const url = `https://wa.me/${whatsappNumber}?text=${encodeURIComponent(message)}`;
    window.open(url, "_blank");
  };
 
  return (
    <Button
      onClick={handleWhatsAppClick}
      className="fixed bottom-4 right-4 sm:bottom-6 sm:right-6 z-50 h-12 w-12 sm:h-16 sm:w-16 rounded-full bg-[#25D366] hover:bg-[#128C7E] text-white shadow-lg hover:shadow-2xl transition-all duration-300 transform hover:scale-110 group whatsapp-pulse whatsapp-float will-change-transform"

      aria-label="Conversar no WhatsApp"

        <WhatsAppIcon />
      </div>
      <span className="sr-only">Conversar no WhatsApp</span>

        Fale Conosco
      </div>
    </Button>
  );
};
